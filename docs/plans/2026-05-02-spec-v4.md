```go
// extractDocMeta returns (name, isTemplate, extendsName) parsed from @doc or @template directives.
// Supports modifier syntax: "@doc Name extends Parent" / "@template Name extends Parent".
// The "extends Parent" suffix is optional; if absent extendsName is "".
// @template name → name from template, isTemplate=true
// @doc name → name from doc, isTemplate=false
func extractDocMeta(directives []ast.Directive) (name string, isTemplate bool, extendsName string) {
	for _, d := range directives {
		if d.Kind == ast.DirectiveTemplate {
			name, extendsName = parseDocArgs(d.Args)
			return name, true, extendsName
		}
	}
	for _, d := range directives {
		if d.Kind == ast.DirectiveDoc {
			name, extendsName = parseDocArgs(d.Args)
			return name, false, extendsName
		}
	}
	return "", false, ""
}

// parseDocArgs splits "<name>" or "<name> extends <parent>" into (name, parent).
// Whitespace-tolerant. If "extends" keyword is absent, parent is "".
func parseDocArgs(args string) (name, parent string) {
	args = strings.TrimSpace(args)
	if args == "" {
		return "", ""
	}
	fields := strings.Fields(args)
	if len(fields) >= 3 && fields[1] == "extends" {
		return fields[0], strings.Join(fields[2:], " ")
	}
	return fields[0], ""
}
```

```go
func mergeExtends(modifier, directive string) string {
	if modifier != "" {
		return modifier

// validateExtendsConflict returns E075 if both modifier and directive set a (different) parent.
func validateExtendsConflict(modifier, directive string, directives []ast.Directive) *ast.Diagnostic {
	if modifier == "" || directive == "" {
		return nil
	}
	if modifier == directive {
		return nil // benign duplicate; deprecation info will still fire
	}
	// find @extends position for range
	pos := ast.Position{}
	for _, d := range directives {
		if d.Kind == ast.DirectiveExtends {
			pos = d.Position
			break
		}
	}
	return &ast.Diagnostic{
		Severity: ast.SeverityError,
		Code:     "E075",
		Message:  fmt.Sprintf("conflicting extends declarations: modifier %q vs directive %q", modifier, directive),
		Range:    ast.Range{Start: pos, End: pos},
	}
}

// validateExtendsDeprecation emits I001 if @extends directive is used.
func validateExtendsDeprecation(directive string, directives []ast.Directive) *ast.Diagnostic {
	if directive == "" {
		return nil
	}
	pos := ast.Position{}
	for _, d := range directives {
		if d.Kind == ast.DirectiveExtends {
			pos = d.Position
			break
		}
	}
	return &ast.Diagnostic{
		Severity: ast.SeverityInfo,
		Code:     "I001",
		Message:  "@extends directive is deprecated; use modifier syntax: @doc <name> extends <parent>",
		Range:    ast.Range{Start: pos, End: pos},
	}
}
```

Re-read the user's spec note: "@import, @let, @const 는 첫 @doc/@template/첫 헤딩(`#`) 보다 위에서만 선언 가능 / 본문 진입 후 file-level 변수 선언 금지". So **file-level only**. Section-scope `@let`/`@const` inside a heading still allowed.

The implementation must distinguish file-level from heading-scope by line position relative to body start. The current `validateImportOrder` only checks `@import` order — not `@const`/`@let`. We need a new validator that also walks `@const`/`@let` and tags only those that are above the first heading/doc/template line yet appear *after* the body start.

Wait — re-reading carefully: "본문 진입 후 file-level 변수 선언 금지". File-level means top-level scope (not inside any heading). So the rule is: any `@import`, `@let`, `@const` that is **at file level** (i.e., its line is before the first heading) must come **before** the first `@doc`/`@template`/heading.

Actually simpler reading: bodyStart = line of first heading OR first `@doc`/`@template` (whichever comes first). After bodyStart, no `@import`/`@const`/`@let` at file-scope. **Section-scope** decls (inside a heading scope, i.e., on a line that is descendant of a heading) are unaffected.

To detect "section scope" reliably: a `@const`/`@let` directive whose line is `>` the first heading's line is inside the heading scope by the same logic the scope tree uses. So:

- `bodyStart = min(first heading line, first @doc line, first @template line)` (1-based; -1 if none)
- For each `@const`/`@let`/`@import` directive D at line `L`:
  - If `bodyStart > 0` and `L > bodyStart` → E007 (file-level decl after body start) **only if** the directive is on a file-level line (i.e., not inside a section that has heading prefix). For `@import`, *always* — already E004 covers it. For `@const`/`@let`, only when **outside any heading scope**.

Simpler operational definition: emit E007 for `@const`/`@let`/`@import` with line `L > bodyStart` **if** there is no heading at level ≥ 1 on a line `≤ L`. But that's just `L < firstHeadingLine` — which contradicts `L > bodyStart`. So actually file-level decls **after** body start can only exist if body start was a `@doc`/`@template` (no heading yet). E.g.,

```md
<!-- @doc x -->
<!-- @let bad = 1 -->   ← E007 (file-level, after @doc)
# X
```

Here line 2 has no preceding heading, but `bodyStart=1` (the `@doc` line). So `L=2 > bodyStart=1` and no heading above → file-level → E007.

vs.

```md
<!-- @doc x -->
# X
## A
<!-- @let inner = 1 -->   ← OK (heading scope; line 4 is inside ## A)
```

Here line 4 has heading `## A` at line 3 above it → in heading scope → no E007.

Algorithm:

```go
func validateFilePrelude(directives []ast.Directive, headings []*ast.Heading) []ast.Diagnostic {
	bodyStart := -1
	for _, d := range directives {
		if d.Kind == ast.DirectiveDoc || d.Kind == ast.DirectiveTemplate {
			bodyStart = d.Position.Line
			break
		}
	}
	if len(headings) > 0 {
		firstH := headings[0].Position.Line
		if bodyStart < 0 || firstH < bodyStart {
			bodyStart = firstH
		}
	}
	if bodyStart < 0 {
		return nil
	}

	// firstHeadingLine for "is in heading scope" check
	firstHeadingLine := -1
	if len(headings) > 0 {
		firstHeadingLine = headings[0].Position.Line
	}

	var diags []ast.Diagnostic
	for _, d := range directives {
		isFilePrelude := d.Kind == ast.DirectiveImport ||
			d.Kind == ast.DirectiveConst ||
			d.Kind == ast.DirectiveLet
		if !isFilePrelude {
			continue
		}
		if d.Position.Line <= bodyStart {
			continue // before body start — fine
		}
		// after body start — only OK if inside a heading scope
		if firstHeadingLine > 0 && d.Position.Line > firstHeadingLine {
			continue // descendant of a heading — section scope — OK
		}
		// file-level decl after body start
		diags = append(diags, ast.Diagnostic{
			Severity: ast.SeverityError,
			Code:     "E007",
			Message:  fmt.Sprintf("file-level declaration after body start: @%s", directiveKindName2(d.Kind)),
			Range:    ast.Range{Start: d.Position, End: d.Position},
		})
	}
	return diags
}

func directiveKindName2(k ast.DirectiveKind) string {
	switch k {
	case ast.DirectiveImport:
		return "import"
	case ast.DirectiveConst:
		return "const"
	case ast.DirectiveLet:
		return "let"
	}
	return "unknown"
}
```

```go
// validateRefsAreAliasOnly enforces rule 3: {{}} and [[]] target tokens must be
// either local variables (handled by scope), property access (alias.field), heading
// symbols (#section), import-alias references (alias#sym), or [[alias]] for links.
// Anything that looks like a raw filesystem path (contains "/" or "./" or "../") is E023.
//
// For [[]] (links), only [[alias]] (with optional #section) is allowed — no raw paths,
// no plain text. Unknown alias raises E024.
//
// For {{}}, raw paths are E023 too. Plain identifiers are accepted (variable lookup).
func validateRefsAreAliasOnly(refs []ast.Reference, imports []ast.Import) []ast.Diagnostic {
	aliases := make(map[string]bool)
	for _, imp := range imports {
		aliases[imp.Alias] = true
	}

	var diags []ast.Diagnostic
	for _, ref := range refs {
		if ref.IsEscaped {
			continue
		}
		// the "head" token to validate: PathPart for {{name}} / {{alias.x}} / {{alias#x}} / [[alias]] / [[alias#x]]
		head := ref.PathPart
		// special case: {{#section}} — head is empty, section refers to current file — fine
		if head == "" && ref.Section != "" {
			continue
		}
		// raw path detection — slashes or dot-prefix
		if strings.Contains(head, "/") || strings.HasPrefix(head, "./") || strings.HasPrefix(head, "../") || strings.HasPrefix(head, ".") {
			kind := "{{}}"
			if ref.IsLink {
				kind = "[[]]"
			}
			diags = append(diags, ast.Diagnostic{
				Severity: ast.SeverityError,
				Code:     "E023",
				Message:  fmt.Sprintf("raw path not allowed in %s; declare with @import alias from <path> first", kind),
				Range:    ast.Range{Start: ref.Position, End: ref.Position},
			})
			continue
		}
		// for [[]] the head MUST be a known alias (no plain identifiers)
		if ref.IsLink {
			if !aliasRe.MatchString(head) {
				diags = append(diags, ast.Diagnostic{
					Severity: ast.SeverityError,
					Code:     "E024",
					Message:  fmt.Sprintf("invalid alias in [[%s]]", ref.Raw),
					Range:    ast.Range{Start: ref.Position, End: ref.Position},
				})
				continue
			}
			if !aliases[head] {
				diags = append(diags, ast.Diagnostic{
					Severity: ast.SeverityError,
					Code:     "E024",
					Message:  fmt.Sprintf("unknown alias in %s; declare with @import alias from <path>", ref.Raw),
					Range:    ast.Range{Start: ref.Position, End: ref.Position},
				})
			}
		}
	}
	return diags
}
```

```go
func BuildDependencyGraphs(ws *workspace.Workspace) DependencyGraphs {
	gExt := DependencyGraph{Edges: make(map[string][]string)}
	gEmb := DependencyGraph{Edges: make(map[string][]string)}

	for _, doc := range ws.DocsByPath {
		docID := doc.Path
		if doc.Name != "" {
			docID = doc.Name
		}

			gExt.Edges[docID] = append(gExt.Edges[docID], doc.ExtendsName)

			}

							targetID = target.Name

					}

					gEmb.Edges[docID] = append(gEmb.Edges[docID], ref.PathPart)

func detectCyclesWithCode(g DependencyGraph, code, label string) []ast.Diagnostic {
	var diags []ast.Diagnostic
	visited := make(map[string]bool)
	inStack := make(map[string]bool)
	var dfs func(node string, path []string)
	dfs = func(node string, path []string) {
		visited[node] = true
		inStack[node] = true
		for _, dep := range g.Edges[node] {
			if !visited[dep] {
				newPath := append(append([]string{}, path...), dep)
				dfs(dep, newPath)
			} else if inStack[dep] {
				cyclePath := append(append([]string{}, path...), dep)
				diags = append(diags, ast.Diagnostic{
					Severity: ast.SeverityError,
					Code:     code,
					Message:  fmt.Sprintf("%s: %s", label, strings.Join(cyclePath, " → ")),
				})
			}
		}
		inStack[node] = false
	}
	for node := range g.Edges {
		if !visited[node] {
			dfs(node, []string{node})
		}
	}
	return diags
}
```
